rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ユーザー
    match /users/{uid} {
      allow read: if true;
      allow create: if request.auth != null && request.auth.uid == uid;
      allow update: if request.auth != null && request.auth.uid == uid;
    }

    // シナリオ
    match /scenarios/{scenarioId} {
      // get: 認証済みなら存在チェック含め許可、未認証でも公開は読める
      allow get: if request.auth != null
                  || resource.data.isPrivate == false;
      // list: 公開は誰でも、非公開は本人のみ
      allow list: if resource.data.isPrivate == false
                   || (request.auth != null && request.auth.uid == resource.data.userId);
      // 認証済みユーザーが自分のデータとして投稿
      allow create: if request.auth != null
                    && request.resource.data.userId == request.auth.uid;
      // 本人: 全フィールド更新・削除可
      // 他ユーザー: bookmarkCountのみ更新可（ブックマークトグル時のバッチ更新）
      allow update: if request.auth != null
                    && (request.auth.uid == resource.data.userId
                        || request.resource.data.diff(resource.data).affectedKeys().hasOnly(["bookmarkCount"]));
      allow delete: if request.auth != null && request.auth.uid == resource.data.userId;

      // クリア報告（サブコレクション）
      match /clears/{clearId} {
        allow read: if true;
        allow create: if request.auth != null
                      && request.resource.data.userId == request.auth.uid;
        allow update, delete: if request.auth != null
                              && request.auth.uid == resource.data.userId;
      }
    }

    // ブックマーク（ドキュメントID = scenarioId_userId）
    match /bookmarks/{bmId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null
                    && request.resource.data.userId == request.auth.uid;
      allow delete: if request.auth != null
                    && resource.data.userId == request.auth.uid;
    }

    // collectionGroupクエリ用（自分のクリア報告横断検索）
    match /{path=**}/clears/{clearId} {
      allow read: if true;
    }
  }
}
